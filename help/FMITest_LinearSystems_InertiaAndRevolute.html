model Reference "Reference solution in pure Modelica" extends Modelica.Icons.Example; Modelica.Mechanics.Rotational.Sources.Torque torque(useSupport=true); Modelica.Blocks.Sources.Sine sine(freqHz=1, amplitude=0.5); inner Modelica.Mechanics.MultiBody.World world(gravityType=Modelica.Mechanics.MultiBody.Types.GravityTypes.UniformGravity); Modelica.Mechanics.MultiBody.Joints.Revolute rev( n={0,0,1}, useAxisFlange=true, phi(fixed=true), w(fixed=true)); Modelica.Mechanics.Rotational.Components.Damper damper(d=0.1); Modelica.Mechanics.MultiBody.Parts.Body body(m=1.0, r_CM={0.5,0,0}); Modelica.Mechanics.Rotational.Components.Inertia inertia( phi(start=0), w(start=0), J=0.05); Modelica.Mechanics.Rotational.Components.Fixed fixed; Modelica.Mechanics.Rotational.Components.IdealGear idealGear(useSupport= true, ratio=3); equation connect(sine.y, torque.tau); connect(world.frame_b, rev.frame_a); connect(body.frame_a, rev.frame_b); connect(fixed.flange, torque.support); connect(fixed.flange, damper.flange_b); connect(fixed.flange, idealGear.support); connect(damper.flange_a, idealGear.flange_a); connect(torque.flange, idealGear.flange_a); connect(idealGear.flange_b, inertia.flange_a); connect(inertia.flange_b, rev.axis); end Reference;
