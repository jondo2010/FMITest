model Reference "Reference solution in pure Modelica" extends Modelica.Icons.Example; Modelica.Mechanics.Rotational.Components.Inertia inertia( J=1.1, w(start=0, fixed=true), phi(start=0, fixed=false), a(fixed=true, start=0)); Modelica.Mechanics.Rotational.Sources.Torque torque1; Modelica.Blocks.Sources.Sine sine( freqHz=1.2, amplitude=2, startTime=0.5); Modelica.Mechanics.Rotational.Components.BearingFriction friction( useSupport=false); Modelica.Mechanics.Rotational.Components.SpringDamper springDamper(c=1000, d=10); Modelica.Mechanics.Rotational.Components.Fixed fixed; Modelica.Mechanics.Rotational.Sources.Torque torque2; Modelica.Blocks.Sources.Constant const(k=2.0); equation connect(sine.y, torque1.tau); connect(friction.flange_a, torque1.flange); connect(inertia.flange_a, friction.flange_b); connect(inertia.flange_b, springDamper.flange_a); connect(springDamper.flange_b, fixed.flange); connect(torque2.flange, torque1.flange); connect(const.y, torque2.tau); end Reference;
